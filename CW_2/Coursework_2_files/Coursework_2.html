def gen_regression_data(num_samples = 10000, ndims=1):
    # generate random x samples for training and test sets
    xTr = torch.rand(num_samples, ndims)
    xTe = torch.rand(int(num_samples * 0.1), ndims)
    
    # construct random w and b vectors
    gt_w = torch.randn(ndims, 1)
    gt_b = torch.randn(1)
    
    # gaussian noise for linear regression
    noise = np.random.rand(num_samples, 1) * 0.2
    test_noise = np.random.rand(int(num_samples * 0.1), 1) * 0.2
    
    # add noise on the labels for the training set
    yTr = xTr @ gt_w + gt_b + noise
    yTe = xTe @ gt_w + gt_b + test_noise
    
    return xTr, xTe, yTr, yTe, gt_w, gt_b

Now we will create our PyTorch model. PyTorch models inherit the class torch.nn.module, and you need to implement the function forward which is equivalent to a forward pass. Usually, you feed in batch of x samples as input and you get batch of outputs, but you could pass other parameters as well if needed. Every torch module will implement two functions. init its constructor, and forward which defines what happens when you call the module.

Note we define two fields of the LinearRegressionModel:

  • self.w: The weight vector of the linear regression model. This is updated automatically by Pytorch in our training loop because we define it as a nn.Parameter (note requires_grad=True). Additionally torch.randn(ndims, 1) gives us an initialization for this weight vector.
  • self.b: The bias of the linear regression model. This is also updated automatically by Pytorch in our training loop.
In [4]:
class LinearRegressionModel(nn.Module):
    def __init__(self, ndims):
        super(LinearRegressionModel, self).__init__()
        """ pytorch optimizer checks for the properties of the model, and if
            the torch.nn.Parameter requires gradient, then the model will update
            the parameters automatically.
        """
        self.w = nn.Parameter(torch.randn(ndims, 1), requires_grad=True)
        self.b = nn.Parameter(torch.randn(1), requires_grad=True)
    
    def forward(self, x):
        return x @ self.w + self.b

def mse_loss(y_pred, y_true):
    square_diff = torch.square((y_pred-y_true))
    mean_error = 0.5 * torch.mean(square_diff)
    return mean_error

def train_regression_model(xTr, yTr, num_epochs, reg_param, lr=1e-2, print_freq=100):
    ndims = xTr.shape[1]
    
    model = LinearRegressionModel(ndims)  # initialize the model
    optimizer = optim.SGD(model.parameters(), lr=lr)  # create an SGD optimizer for the model parameters
    
    for epoch in range(num_epochs):
        # need to zero the gradients in the optimizer so we don't
        # use the gradients from previous iterations
        optimizer.zero_grad()  
        pred = model.forward(xTr)  # compute model predictions
        loss = mse_loss(pred, yTr) + reg_param * torch.norm(model.w)
        loss.backward()  # compute the gradient wrt loss
        optimizer.step()  # performs a step of gradient descent
        if (epoch + 1) % print_freq == 0:
            print('epoch {} loss {}'.format(epoch+1, loss.item()))
    
    return model  # return trained model

def genrandomdata(n=100,b=0.):

def hinge_loss(y_pred, y_true):
    nonzero_diff = torch.clamp(1-y_pred*y_true, min=0)
    return torch.sum(nonzero_diff)

def primalSVM(xTr, yTr, num_epochs=1000, C=1):
    # initialize LinearSVM model
    model = LinearSVM(xTr.shape[1])
    # optimizer for parameters
    optimizer = optim.SGD(model.parameters(), lr=1e-2)
    for i in range(num_epochs):
        index = torch.randperm(xTr.shape[0])
        xTr = xTr[index]
        yTr = yTr[index]
        for j in range(xTr.shape[0]):
            optimizer.zero_grad()
            xt, yt = xTr[j], yTr[j]
            pred = model.forward(xt)
            # regularization term
            loss = hinge_loss(pred, yt)*C
            loss.backward()
            optimizer.step()

    svmclassify = lambda x:model.forward(x).flatten()
    return svmclassify

def spiraldata(N=300):
    r = np.linspace(1,2*np.pi,N)
    xTr1 = np.array([np.sin(2.*r)*r, np.cos(2*r)*r]).T
    xTr2 = np.array([np.sin(2.*r+np.pi)*r, np.cos(2*r+np.pi)*r]).T
    xTr = np.concatenate([xTr1, xTr2], axis=0)
    yTr = np.concatenate([np.ones(N), -1 * np.ones(N)])
    xTr = xTr + np.random.randn(xTr.shape[0], xTr.shape[1])*0.2
    
    xTe = xTr[::2,:]
    yTe = yTr[::2]
    xTr = xTr[1::2,:]
    yTr = yTr[1::2]
    
    xTr = torch.tensor(xTr).float()
    yTr = torch.tensor(yTr).float()
    xTe = torch.tensor(xTe).float()
    yTe = torch.tensor(yTe).float()
    
    vals, indices = torch.max(xTr, dim=0, keepdim=True)
    xTr /= (vals * 2.0)
    vals, indices = torch.max(xTe, dim=0, keepdim=True)
    xTe /= (vals * 2.0)
    
    return xTr,yTr,xTe,yTe

Implementing a kernelized SVM

For a data set as complex as the spiral data set, you will need a more complex classifier. First implement the kernel function 

	computeK(kernel_type,X,Z,kpar)
It takes as input a kernel type kernel_type and two data sets XRndXRnd and ZRmdZRmd and outputs a kernel matrix KRnmKRnm. The last input, kpar specifies the kernel parameter (e.g. the inverse kernel width in the RBF case or the degree pp in the polynomial case.)
  1. For the linear kernel (kernel_type='linear') svm, use k(x,z)=xTzk(x,z)=xTz
  2. For the radial basis function kernel (kernel_type='rbf') svm use k(x,z)=exp(||xz||2)k(x,z)=exp(||xz||2) (gamma is a hyperparameter, passed as the value of kpar)
  3. For the polynomial kernel (kernel_type='poly') use k(x,z)=(xTz+1)dk(x,z)=(xTz+1)d (d is the degree of the polymial, passed as the value of kpar)

You can use the function l2distance as a helperfunction, which is located in defined in one of your starter files l2distance.py.

def computeK(kernel_type, X, Z, kpar=0):
    """
    function K = computeK(kernel_type, X, Z)
    computes a matrix K such that Kij=k(x,z);
    for three different function linear, rbf or polynomial.
    
    Input:
    kernel_type: either 'linear','polynomial','rbf'
    X: n input vectors of dimension d (nxd);
    Z: m input vectors of dimension d (mxd);
    kpar: kernel parameter (inverse kernel width gamma in case of RBF, degree in case of polynomial)
    
    OUTPUT:
    K : nxm kernel Torch float tensor
    """
    assert kernel_type in ["linear","polynomial","poly","rbf"], "Kernel type %s not known." % kernel_type
    assert X.shape[1] == Z.shape[1], "Input dimensions do not match"
    
    # linear type
    if kernel_type == "linear":
        K = X @ Z.T
    
    # rbf type
    elif kernel_type == "rbf":
        dist_sq = l2distance(X, Z)**2
        K = torch.exp(-kpar*dist_sq) 
    
    # polynomial type
    else:
        mat = X @ Z.T + 1
        K = torch.pow(mat, kpar)
    
    return K

Remember that the SVM optimization has the following dual formulation: (1) min1,,n12i,jijyiyjKijni=1is.t.0iCni=1iyi=0.min1,,n12i,jijyiyjKiji=1nis.t.0iCi=1niyi=0. This is equivalent to solving for the SVM primal (2) L(w,b)=Cni=1max(1yi(w(xi)+b),0)+||w||22L(w,b)=Ci=1nmax(1yi(w(xi)+b),0)+||w||22 where w=ni=1yii(xi)w=i=1nyii(xi) and Kij=k(xi,xj)=(xi)(xj)Kij=k(xi,xj)=(xi)(xj), for some mapping ()(). However, after a change of variable, with i=iyii=iyi and RnRn, (2) can be rewritten as follows (see https://arxiv.org/pdf/1404.1066.pdf for details): min,b12K+C2ni=1[max(1yi(ki+b),0)]2min,b12K+C2i=1n[max(1yi(ki+b),0)]2 where kiki is the kernel matrix row corresponding to the ith training example. Notice that there are two relaxations: 1. the ii are unconstrained, in contrast to ii in (1), which must satisfy 0iC0iC; and 2. the squared hinge loss is used in place of the more common absolute hinge loss.

class KernelizedSVM(nn.Module):
    def __init__(self, dim, kernel_type, kpar=0):
        super(KernelizedSVM, self).__init__()
        # Initialization
        self.kpar = kpar
        self.kernel_type = kernel_type
        self.beta = nn.Parameter(torch.zeros(dim, 1), requires_grad=True)
        self.b = nn.Parameter(torch.zeros(1), requires_grad=True)
        
    def forward(self, xTr, x):
        # Create kernel
        K = computeK(self.kernel_type, xTr, x, self.kpar)
        h_x = self.beta.T @ K + self.b
        return h_x

  • torch.square(x) Returns a new tensor with the square of the elements of input.

def kernelsvm_loss(kernelizedSVM, kernel_mat, yTr, C):
    # compute hinge loss square
    beta, b = kernelizedSVM.beta, kernelizedSVM.b
    diff = 1 - yTr*(beta.T@kernel_mat + b)
    hinge_loss_sq = torch.sum(torch.relu(diff)**2)*C/2
    # compute cumulative loss
    cumulative_loss = hinge_loss_sq + (beta.T @ kernel_mat @ beta)/2
    return cumulative_loss

def dualSVM(xTr, yTr, kernel_type, num_epochs=100, C=1, lmbda=0, lr=1e-3):
    dim = len(xTr)
    # initialize kernelsvm model
    kernelsvm = KernelizedSVM(dim, kernel_type, lmbda)
    # optimizer for parameters
    optimizer = optim.SGD(kernelsvm.parameters(), lr=lr)
    for i in range(num_epochs):
        # reset gradient
        optimizer.zero_grad()
        K_mat = computeK(kernel_type, xTr, xTr, lmbda)
        kernel_svm_loss = kernelsvm_loss(kernelsvm, K_mat, yTr, C)
        kernel_svm_loss.backward()
        optimizer.step()
    # return lambda function
    svmclassify = lambda x:kernelsvm.forward(xTr, x).flatten()
    return svmclassify

def cross_validation(xTr,yTr,xValid,yValid,ktype,CList,lmbdaList,lr_List):
    """
    function bestC,bestLmbda,ErrorMatrix = cross_validation(xTr,yTr,xValid,yValid,ktype,CList,lmbdaList);
    Use the parameter search to find the optimal parameter,
    Individual models are trained on (xTr,yTr) while validated on (xValid,yValid)
    
    Input:
        xTr      | training data (nxd)
        yTr      | training labels (nx1)
        xValid   | training data (mxd)
        yValid   | training labels (mx1)
        ktype    | the type of kernelization: 'rbf','polynomial','linear'
        CList    | The list of values to try for the SVM regularization parameter C (ax1)
        lmbdaList| The list of values to try for the kernel parameter lmbda- degree for poly, inverse width for rbf (bx1)
        lr_list  | The list of values to try for the learning rate of our optimizer
    
    Output:
        bestC      | the best C parameter
        bestLmbda  | the best Lmbda parameter
        bestLr     | the best Lr parameter
        ErrorMatrix| the test error rate for each given (C, Lmbda Lr) tuple when trained on (xTr,yTr) and tested on (xValid,yValid)
    """
    # create a 3-d matrix store error rate
    c, n, m = len(CList), len(lmbdaList), len(lr_List)
    ErrorMatrix = np.zeros((c, n, m))
    # iterations for all cases
    for i in range(c):
        C_i = CList[i]
        for j in range(n):
            lmbda_j = lmbdaList[j]
            for k in range(m):
                lr_k = lr_List[k]
                kernel_svm = dualSVM(xTr, yTr, ktype, 25, C_i, lmbda_j, lr_k)
                y_pred = kernel_svm(xValid).sign()
                error_id = (y_pred != yValid).float()
                error_rate = torch.mean(error_id)
                ErrorMatrix[i][j][k] = error_rate
    # get index for the least error rate
    least_error = np.min(ErrorMatrix)
    bestC_id = np.where(ErrorMatrix==least_error)[0]
    bestLmbda_id = np.where(ErrorMatrix==least_error)[1]
    bestLr_id = np.where(ErrorMatrix==least_error)[2]
    bestC,bestLmbda,bestLr = CList[bestC_id][0], lmbdaList[bestLmbda_id][0], lr_List[bestLr_id][0]
    return bestC,bestLmbda,bestLr,ErrorMatrix
